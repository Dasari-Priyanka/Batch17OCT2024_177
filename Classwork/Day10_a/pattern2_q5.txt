#include<stdio.h>

int main()
{
    int N,i;
    scanf("%d",&N);
    for(int i=1;i<=N;i++)
    {
        for(int j=N;j>i;j--)
            printf(" ");
        for(int j=1;j<=i;j++)
            printf("%c",'1'+j-1);
    
    
        for(int j=i-1;j>=1;j--)
            printf("%c",'1'+j-1);
        
        printf("\n");
        
    }
    return 0;
}
